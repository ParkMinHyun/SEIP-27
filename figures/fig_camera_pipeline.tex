% Logical Draft and deferred post-processing paths.
% Included from 2_3_draft_sequence.tex; contains the full figure environment
% (Figure~\ref{fig:draft-pipeline}).
%
% This is a logical dataflow view, not an execution-timing diagram. The
% Draft Sequence publishes an early capture result, while post-processing
% is deferred until the camera application switches to the background.
% If post-processing fails or is interrupted, the published Draft image is
% retained as the recovery result instead of being replaced.
\begin{figure}[!ht]
  \centering
  \begin{tikzpicture}[
    font=\scriptsize,
    >=latex,
    draftbox/.style={draw=blue!45!black, fill=blue!14, rounded corners=1pt,
      align=center},
    finalbox/.style={draw=black!65, fill=gray!14, rounded corners=1pt,
      align=center},
    sharedbox/.style={draw=black!55, fill=gray!10, rounded corners=1pt,
      align=center},
    flow/.style={->, thick, black!65, rounded corners=2pt},
    recover/.style={->, dashed, thick, red!45!black, rounded corners=2pt},
    note/.style={text=black!55, align=left}
  ]

    % ---- shutter event (camera glyph) ----
    \draw[fill=gray!15, draw=black!70] (0.36,2.26) rectangle (0.62,2.40);
    \draw[fill=gray!15, draw=black!70, rounded corners=2pt]
      (0.13,1.72) rectangle (0.97,2.28);
    \draw[fill=white, draw=black!70] (0.55,2.00) circle (0.17);
    \draw[fill=gray!45, draw=black!70] (0.55,2.00) circle (0.085);
    \node[anchor=north, font=\scriptsize\bfseries] at (0.55,1.62) {shutter};

    % ---- shared collection stage ----
    \draw[flow] (1.00,2.00) -- (1.28,2.00);
    \draw[sharedbox] (1.30,1.52) rectangle (2.60,2.48);
    \node[align=center] at (1.95,2.00) {collect\\frames \&\\metadata};

    % ---- fork into the two processing paths ----
    \draw[thick, black!65] (2.60,2.00) -- (2.82,2.00);
    \draw[flow]  (2.82,2.00) |- (3.10,2.95);
    \draw[flow] (2.82,2.00) |- (3.10,1.05);

    % ---- Draft path (capture-critical) ----
    \draw[draftbox] (3.12,2.67) rectangle (5.02,3.23);
    \node at (4.07,2.95) {Draft Sequence};
    \draw[black!65, thick] (5.02,2.95) -- (5.53,2.95);

    % Schematic Draft image.
    \fill[white] (5.55,2.69) rectangle (6.17,3.21);
    \fill[gray!50] (5.57,2.71) -- (5.72,2.99) -- (5.87,2.71) -- cycle;
    \fill[gray!50] (5.84,2.71) -- (6.00,3.03) -- (6.15,2.71) -- cycle;
    \draw[fill=gray!65, draw=white, line width=0.4pt]
      (5.70,2.69) .. controls (5.70,2.86) and (6.02,2.86) .. (6.02,2.69)
      -- cycle;
    \draw[fill=gray!65, draw=white, line width=0.4pt]
      (5.86,2.93) circle (0.065);
    \draw[black!60] (5.55,2.69) rectangle (6.17,3.21);
    \node[anchor=north, text=black!70] at (5.86,2.62) {Draft image};

    \draw[flow] (6.19,2.95) -- (6.76,2.95);

    % ---- user-visible result ----
    \draw[draw=black!60, fill=blue!4, rounded corners=3pt]
      (6.78,2.60) rectangle (8.48,3.28);
    \node at (7.63,2.94) {shown to user};

    % ---- final path ----
    \draw[finalbox] (3.12,0.73) rectangle (6.30,1.37);
    \node at (4.71,1.05) {Post-processing};

    \draw[black!65, thick] (6.30,1.05) -- (6.53,1.05);

    % Schematic final image.
    \fill[white] (6.55,0.79) rectangle (7.17,1.31);
    \fill[blue!12] (6.56,0.80) -- (6.72,1.10) -- (6.88,0.80) -- cycle;

    \fill[blue!22] (6.66,1.16) circle (0.075);
    \fill[blue!14] (6.85,1.22) circle (0.055);

    \draw[flow] (7.19,1.05) -- (7.60,1.05) -- (7.60,2.58);

    \draw[recover] (5.86,1.38) -- (5.86,2.28);
    \node[anchor=east, align=right, text=red!45!black] at (5.82,1.82)

  \end{tikzpicture}
  \caption{Draft Sequence, post-processing, and recovery.}

% Draft Sequences piling up under parallel capture (three shots).
% Included from 2_2_parallel_capture.tex (Figure~\ref{fig:overlap}).
% Requires TikZ with the `patterns', `decorations.pathreplacing', and
% `positioning' libraries, plus pifont for the check mark.
%
% The figure intentionally abstracts away internal Draft workloads. Each
% capture first collects frames and metadata. These collection periods can
% overlap, whereas Draft Sequences run one at a time in capture order. The
% resulting worker wait grows across the burst until capture i+2 misses
% its deadline despite having the same Draft Sequence duration.
\begin{tikzpicture}[font=\small,
  collectstage/.style={draw=black!55, fill=gray!14},
  draftstage/.style={draw=blue!45!black, fill=blue!14},
  queued/.style={pattern=north east lines, pattern color=red!45, draw=red!35},
  overrun/.style={draw=red!80!black, fill=red!65},
  swatch/.style={draw, minimum width=8pt, minimum height=6pt, inner sep=0pt},
  ddl/.style={red, dashed, thick},
  sig/.style={->, black!75}]

  % ---------- legend ----------
  \node[anchor=east, inner sep=2pt] (lt2) at (9.70, 5.40)
    {\footnotesize Draft-worker wait};
  \node[swatch, queued, left=0mm of lt2] (lg2) {};
  \node[anchor=east, inner sep=2pt, left=7mm of lg2] (lt1)
    {\footnotesize frames \& metadata};
  \node[swatch, collectstage, left=0mm of lt1] (lg1) {};

  % ---------- capture i ----------
  \node[anchor=east] at (-0.2, 3.925) {capture $i$};
  \draw[very thick] (0, 3.55) -- (0, 4.30);
  \node[anchor=south] at (0, 4.32) {\scriptsize request accepted};
  \draw[collectstage] (0.05, 3.70) rectangle (0.90, 4.15);
  \node at (0.475, 3.925) {\scriptsize collect};
  \draw[draftstage] (0.90, 3.70) rectangle (3.50, 4.15);
  \node at (2.20, 3.925) {\scriptsize Draft Sequence};

  \draw[densely dotted, gray] (3.50, 3.925) -- (7.30, 3.925);
  \node[fill=white, inner sep=1.5pt] at (5.40, 3.925)
    {\footnotesize\textcolor{green!45!black}{\ding{51} in time}};
  \draw[ddl] (7.30, 3.55) -- (7.30, 4.30);
  \node[red, anchor=south] at (7.30, 4.32) {deadline $i$};

  % HAL releases the next request before capture i's collection and Draft
  % work have completed.
  \fill[black!60] (0.35, 3.68) circle (1.3pt);
  \draw[sig] (0.35, 3.66) -- (0.50, 2.87);
  \node[anchor=west, black!75] at (0.40, 3.35)
    {\scriptsize\texttt{captureAvailable}};

  % ---------- capture i+1 ----------
  \node[anchor=east] at (0.3, 2.425) {capture $i{+}1$};
  \draw[very thick] (0.50, 2.05) -- (0.50, 2.80);
  \draw[collectstage] (0.55, 2.20) rectangle (1.40, 2.65);
  \node at (0.975, 2.425) {\scriptsize collect};
  \draw[queued] (1.40, 2.20) rectangle (3.50, 2.65);
  \node[fill=white, inner sep=1.5pt] at (2.45, 2.425)
    {\footnotesize\textcolor{red!60!black}{waiting}};
  \draw[draftstage] (3.50, 2.20) rectangle (6.10, 2.65);
  \node at (4.80, 2.425) {\scriptsize Draft Sequence};

  \draw[densely dotted, gray] (6.10, 2.425) -- (7.80, 2.425);
  \node[fill=white, inner sep=1.5pt] at (6.95, 2.425)
    {\footnotesize\textcolor{green!45!black}{\ding{51} in time}};
  \draw[ddl] (7.80, 2.05) -- (7.80, 2.80);
  \node[red, anchor=south] at (7.80, 2.82) {deadline $i{+}1$};

  \fill[black!60] (0.85, 2.18) circle (1.3pt);
  \draw[sig] (0.85, 2.16) -- (1.00, 1.37);

  % ---------- capture i+2 ----------
  \node[anchor=east] at (0.8, 0.925) {capture $i{+}2$};
  \draw[very thick] (1.00, 0.55) -- (1.00, 1.30);
  \draw[collectstage] (1.05, 0.70) rectangle (1.90, 1.15);
  \node at (1.475, 0.925) {\scriptsize collect};
  \draw[queued] (1.90, 0.70) rectangle (6.10, 1.15);
  \node[fill=white, inner sep=1.5pt] at (4.00, 0.925)
    {\footnotesize\textcolor{red!60!black}{waiting}};
  \draw[draftstage] (6.10, 0.70) rectangle (8.70, 1.15);

  % The same Draft duration now extends beyond capture i+2's deadline.
  \draw[overrun] (8.30, 0.70) rectangle (8.70, 1.15);
  \node at (7.20, 0.925) {\scriptsize Draft Sequence};
  \draw[ddl] (8.30, 0.55) -- (8.30, 1.30);
  \node[red, anchor=south] at (8.30, 1.32) {deadline $i{+}2$};
  \node[red!70!black, anchor=west, align=left] at (8.80, 0.925)
    {\footnotesize\bfseries Capture\\[-2pt]\footnotesize\bfseries Timeout};

  % ---------- ordered Draft handoffs ----------
  \draw[dashed, gray, ->] (3.50, 3.67) -- (3.50, 2.68);
  \draw[dashed, gray, ->] (6.10, 2.17) -- (6.10, 1.18);

  % ---------- budget braces for the failing capture ----------
  \draw[decorate, decoration={brace, mirror, amplitude=4pt}]
    (1.00, 0.50) -- (6.10, 0.50);
  \node[anchor=north] at (3.55, 0.28) {\footnotesize pre-Draft budget consumed};
  \draw[decorate, decoration={brace, mirror, amplitude=4pt}]
    (6.10, 0.50) -- (8.30, 0.50);
  \node[anchor=north] at (7.20, 0.28) {\footnotesize remaining budget};

  % ---------- time axis ----------
  \draw[->] (-1.3, -0.4) -- (9.7, -0.4) node[anchor=north east] {time};

\end{tikzpicture}
